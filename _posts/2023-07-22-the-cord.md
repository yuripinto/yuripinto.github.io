Sub CreateOrgChart()
    Dim wsData As Worksheet, wsChart As Worksheet
    Dim rng As Range, cell As Range
    Dim empList As Object, shapeList As Object
    Dim emp As Object
    Dim shp As Shape, conn As Shape
    Dim empName As String, bossName As String, desc As String
    Dim topPos As Double, leftPos As Double
    Dim rootBoss As String
    Dim rowIndex As Integer, colIndex As Integer

    ' Set worksheets
    Set wsData = ThisWorkbook.Sheets("Sheet1")
    On Error Resume Next
    Set wsChart = ThisWorkbook.Sheets("OrgChart")
    If wsChart Is Nothing Then
        Set wsChart = ThisWorkbook.Sheets.Add
        wsChart.Name = "OrgChart"
    Else
        wsChart.Cells.Clear
    End If
    On Error GoTo 0

    ' Using Collection instead of Dictionary
    Set empList = New Collection
    Set shapeList = New Collection

    ' Read data from Sheet1
    Set rng = wsData.Range("A2:C" & wsData.Cells(wsData.Rows.Count, 1).End(xlUp).Row)
    
    ' Populate employee structure
    On Error Resume Next
    For Each cell In rng.Columns(1).Cells
        empName = Trim(cell.Value)
        bossName = Trim(cell.Offset(0, 1).Value)
        desc = Trim(cell.Offset(0, 2).Value)

        If empName <> "" Then
            Set emp = Nothing
            Err.Clear
            Set emp = empList(empName)
            If Err.Number <> 0 Then
                Set emp = New Collection
                emp.Add bossName, "Boss"
                emp.Add desc, "Desc"
                emp.Add New Collection, "Subordinates"
                empList.Add emp, empName
            Else
                emp.Remove "Boss"
                emp.Add bossName, "Boss"
                emp.Remove "Desc"
                emp.Add desc, "Desc"
            End If

            ' Add employee to boss's subordinates list
            If bossName <> "" Then
                Set emp = Nothing
                Err.Clear
                Set emp = empList(bossName)
                If Err.Number <> 0 Then
                    Set emp = New Collection
                    emp.Add "", "Boss"
                    emp.Add "", "Desc"
                    emp.Add New Collection, "Subordinates"
                    empList.Add emp, bossName
                End If
                emp.Item("Subordinates").Add empName
            End If
        End If
    Next cell
    On Error GoTo 0

    ' Find root boss (the one with no boss)
    For Each key In empList
        If key("Boss") = "" Then
            rootBoss = empList.Key(empList.Count) ' Get first top-level boss
            Exit For
        End If
    Next key

    ' Positioning variables
    topPos = 50
    leftPos = 200

    ' Recursive function to draw employees
    Call DrawOrgChart(wsChart, empList, shapeList, rootBoss, leftPos, topPos, 300, 80)

    ' Clean up
    Set empList = Nothing
    Set shapeList = Nothing
End Sub

Sub DrawOrgChart(ws As Worksheet, empList As Collection, shapeList As Collection, empName As String, x As Double, y As Double, hSpacing As Double, vSpacing As Double)
    Dim shp As Shape
    Dim bossShape As Shape
    Dim emp As Collection
    Dim subEmp As Variant
    Dim newX As Double, newY As Double
    Dim i As Integer, count As Integer
    Dim conn As Shape

    ' Create employee shape
    Set shp = ws.Shapes.AddShape(msoShapeRectangle, x, y, 100, 40)
    shp.TextFrame2.TextRange.Text = empName & vbCrLf & empList(empName)("Desc")
    shp.TextFrame2.TextRange.Font.Size = 10
    shp.Fill.ForeColor.RGB = RGB(173, 216, 230) ' Light blue
    shp.Line.ForeColor.RGB = RGB(0, 0, 0) ' Black border
    shapeList.Add shp, empName

    ' If employee has a boss, draw a connector
    If empList(empName)("Boss") <> "" Then
        Set bossShape = shapeList(empList(empName)("Boss"))
        Set conn = ws.Shapes.AddConnector(msoConnectorStraight, 0, 0, 0, 0)
        conn.ConnectorFormat.BeginConnect bossShape, 3
        conn.ConnectorFormat.EndConnect shp, 1
        conn.Line.ForeColor.RGB = RGB(0, 0, 0) ' Black line
    End If

    ' Get subordinates
    count = empList(empName)("Subordinates").Count
    If count > 0 Then
        newX = x - ((count - 1) * hSpacing / 2) ' Center subordinates
        newY = y + vSpacing

        i = 0
        For Each subEmp In empList(empName)("Subordinates")
            Call DrawOrgChart(ws, empList, shapeList, subEmp, newX + (i * hSpacing), newY, hSpacing / 1.5, vSpacing)
            i = i + 1
        Next subEmp
    End If
End Sub
