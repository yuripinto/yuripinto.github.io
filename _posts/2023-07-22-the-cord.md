Sub CreateOrgChart()
    Dim wsData As Worksheet, wsChart As Worksheet
    Dim rng As Range, cell As Range
    Dim dict As Object, emp As Object
    Dim shapeDict As Object
    Dim shp As Shape, conn As Shape
    Dim empName As String, bossName As String, desc As String
    Dim topPos As Double, leftPos As Double
    Dim rowIndex As Integer, colIndex As Integer
    
    ' Set worksheets
    Set wsData = ThisWorkbook.Sheets("Sheet1")
    On Error Resume Next
    Set wsChart = ThisWorkbook.Sheets("OrgChart")
    If wsChart Is Nothing Then
        Set wsChart = ThisWorkbook.Sheets.Add
        wsChart.Name = "OrgChart"
    Else
        wsChart.Cells.Clear
    End If
    On Error GoTo 0

    ' Dictionary to store employee hierarchy
    Set dict = CreateObject("Scripting.Dictionary")
    Set shapeDict = CreateObject("Scripting.Dictionary")

    ' Read data from Sheet1
    Set rng = wsData.Range("A2:C" & wsData.Cells(wsData.Rows.Count, 1).End(xlUp).Row)
    
    ' Populate dictionary with employee-boss mapping
    For Each cell In rng.Columns(1).Cells
        empName = Trim(cell.Value)
        bossName = Trim(cell.Offset(0, 1).Value)
        desc = Trim(cell.Offset(0, 2).Value)
        
        If empName <> "" Then
            If Not dict.exists(empName) Then
                Set emp = CreateObject("Scripting.Dictionary")
                emp("Boss") = bossName
                emp("Desc") = desc
                emp("Subordinates") = CreateObject("Scripting.Dictionary")
                dict.Add empName, emp
            Else
                dict(empName)("Boss") = bossName
                dict(empName)("Desc") = desc
            End If
            
            ' Add employee to boss's subordinates list
            If bossName <> "" Then
                If Not dict.exists(bossName) Then
                    Set emp = CreateObject("Scripting.Dictionary")
                    emp("Boss") = ""
                    emp("Desc") = ""
                    emp("Subordinates") = CreateObject("Scripting.Dictionary")
                    dict.Add bossName, emp
                End If
                dict(bossName)("Subordinates").Add empName, 1
            End If
        End If
    Next cell

    ' Find the top-level boss (the one with no boss)
    Dim rootBoss As String
    For Each key In dict.keys
        If dict(key)("Boss") = "" Then
            rootBoss = key
            Exit For
        End If
    Next key

    ' Positioning variables
    topPos = 50
    leftPos = 200

    ' Recursive function to draw employees
    Call DrawOrgChart(wsChart, dict, shapeDict, rootBoss, leftPos, topPos, 300, 80)

    ' Clean up
    Set dict = Nothing
    Set shapeDict = Nothing
End Sub

Sub DrawOrgChart(ws As Worksheet, dict As Object, shapeDict As Object, empName As String, x As Double, y As Double, hSpacing As Double, vSpacing As Double)
    Dim shp As Shape
    Dim bossShape As Shape
    Dim emp As Object
    Dim subEmp As Variant
    Dim newX As Double, newY As Double
    Dim i As Integer, count As Integer
    Dim conn As Shape

    ' Create employee shape
    Set shp = ws.Shapes.AddShape(msoShapeRectangle, x, y, 100, 40)
    shp.TextFrame2.TextRange.Text = empName & vbCrLf & dict(empName)("Desc")
    shp.TextFrame2.TextRange.Font.Size = 10
    shp.Fill.ForeColor.RGB = RGB(173, 216, 230) ' Light blue
    shp.Line.ForeColor.RGB = RGB(0, 0, 0) ' Black border
    shapeDict.Add empName, shp

    ' If employee has a boss, draw a connector
    If dict(empName)("Boss") <> "" Then
        Set bossShape = shapeDict(dict(empName)("Boss"))
        Set conn = ws.Shapes.AddConnector(msoConnectorStraight, 0, 0, 0, 0)
        conn.ConnectorFormat.BeginConnect bossShape, 3
        conn.ConnectorFormat.EndConnect shp, 1
        conn.Line.ForeColor.RGB = RGB(0, 0, 0) ' Black line
    End If

    ' Get subordinates
    count = dict(empName)("Subordinates").Count
    If count > 0 Then
        newX = x - ((count - 1) * hSpacing / 2) ' Center subordinates
        newY = y + vSpacing

        i = 0
        For Each subEmp In dict(empName)("Subordinates").keys
            Call DrawOrgChart(ws, dict, shapeDict, subEmp, newX + (i * hSpacing), newY, hSpacing / 1.5, vSpacing)
            i = i + 1
        Next subEmp
    End If
End Sub
